001  (ns aoc13
002    (:require [clojure.string :as string]))
003  
004  (defn parse-button [line]
005    (let [[_ button x y] (re-matches #"Button (\w): X([+-]\d+), Y([+-]\d+)" line)]
006      {button [(Integer/parseInt x) (Integer/parseInt y)]}))
007  
008  (defn parse-prize [line]
009    (let [[_ x y] (re-matches #"Prize.*X=([+-]?\d+).*Y=([+-]?\d+)" line)]
010      {"X" (Integer/parseInt x) "Y" (Integer/parseInt y)}))
011  
012  (defn parse-machine [lines]
013    (prn 'parse-machine)
014    (let [buttons (map parse-button (take 2 lines))
015          prize (parse-prize (last lines))
016          buttons (apply merge buttons)
017          machine (merge buttons prize)]
018      {:xa (first (machine "A")) :ya (second (machine "A"))
019       :xb (first (machine "B")) :yb (second (machine "B"))
020       :x (machine "X") :y (machine "Y")}))
021  
022  (defn solve-machine
023    ([machine]
024     (solve-machine machine 0))
025  
026    ([{:keys [xa ya xb yb x y]} offset]
027     (let [x (+ x offset)
028           y (+ y offset)
029           a (/ (- (* xb y) (* yb x)) (- (* xb ya) (* xa yb)))
030           b (/ (- x (* xa a)) xb)]
031       (if (and (int? a) (int? b)) (+ (* 3 a) b) 0)
032       )))
033  
034  (defn solve
035    ([filename]
036     (solve filename 0))
037  
038    ([filename offset]
039     (loop [lines (-> filename slurp (string/split-lines))
040            sum 0]
041       (if (empty? lines)
042         sum
043         (let [tokens (-> (take 3 lines)
044                           parse-machine
045                           (solve-machine offset))]
046           (recur (drop 4 lines) (+ sum tokens))))
047       )))
